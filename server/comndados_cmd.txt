## Ver los paquetes instalados en el entorno virtual
pip list --not-required
# Generar el archivo requirements.txt
pip list --not-required --format=freeze > requirements.txt

## Crear la imagende de docker # el punto es para crear el docker en el directorio actual
docker build -t server_bigdata .

# Ejecutar el script Flask en el contenedor
docker run -d -p 5002:5002 --name contenedor_server server_bigdata python app_server.py

#parar la ejecuciond ela imagen de decker
docker stop contenedor_server

# crear red 
docker network create mynetwork

docker run -d -p 5002:5002 --network=mynetwork --name contenedor_server server_bigdata python app_server.py
